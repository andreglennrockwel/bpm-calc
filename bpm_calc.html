    <script>

        function calculateTimes() {

            const bpm = document.getElementById('bpmInput').value;

            const msPerBeat = 60000 / bpm;

            

            // Define all the note divisions down to 256th notes

            const noteTypes = [

                { name: 'Whole Note', division: 1 },

                { name: 'Half Note', division: 2 },

                { name: 'Quarter Note', division: 4 },

                { name: 'Eighth Note', division: 8 },

                { name: 'Sixteenth Note', division: 16 },

                { name: 'Thirty-second Note', division: 32 },

                { name: 'Sixty-fourth Note', division: 64 },

                { name: 'One Hundred Twenty-eighth Note', division: 128 },

                { name: 'Two Hundred Fifty-sixth Note', division: 256 }

            ];


            // Clear previous results

            const tbody = document.querySelector('#resultsTable tbody');

            tbody.innerHTML = '';


            // Calculate and display the times

            noteTypes.forEach(note => {

                const timeInMs = msPerBeat / (note.division / 4); // Divide the beat time by the note division

                const row = `<tr>

                                <td>${note.name}</td>

                                <td>${timeInMs.toFixed(2)} ms</td>

                             </tr>`;

                tbody.innerHTML += row;

            });

        }

    </script>
